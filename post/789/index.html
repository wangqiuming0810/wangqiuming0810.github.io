A. Tokitsukaze and All Zero Sequence

  1. 如果当前这个序列有0，那么最优策略，每一个非0的数都可以通过一步的操作，把非0变成0。

  2. 不存在0，但是存在两个相等的数，可以先通过一次操作把两个相同的数其中一个变成0，然后就跟1操作一样。

  3. 前两种情况都没有，可以通过一次操作将数变成2情况。

代码：

#include<bits/stdc++.h>
using namespace std;
#define int long long
typedef long long ll;
typedef pair<int,int> PII;
const int N = 2e6 + 7;
const int mod = 1e9 + 7;
const int MOD = 998244353;

void solve(){
	int n;
	cin>>n;
	vector<int> a(n,0);
	for(int i=0;i<n;i++){
		cin>>a[i];
	}
	sort(a.begin(),a.end());
	if(a[0]==0){
		int ans=0;
		for(int i=1;i<n;i++){
			if(a[i]>0)ans++;
		}
		cout<<ans<<endl;
	}else{
		int ok=0;
		for(int i=0;i<n-1;i++){
			if(a[i]==a[i+1]) ok=1;
		}
		if(ok) cout<<n<<endl;
		else cout<<n+1<<endl;
	}
	
}
signed main(){
	int _=1;
	cin>>_;
	while(_--) solve();
	return 0;
}

B.Tokitsukaze and Good 01-String

求最小的操作次数肯定是很简单的，两个两个改成一样就可以了。所以两个两个字符不一样答案就要+1；那么最少的串，取决于没有操作的字符，其他的操作了的字符就可以全部改成一样的,并不会影响答案，所以求没有操作的字符连接起来看有多少段。

代码：

#include<bits/stdc++.h>
using namespace std;
#define int long long
typedef long long ll;
typedef pair<int,int> PII;
const int N = 2e6 + 7;
const int mod = 1e9 + 7;
const int MOD = 998244353;

void solve(){
	int n;
	cin>>n;
	string s;
	cin>>s;
	vector<char> ans;
	int res=0;
	for(int i=0;i<n;i+=2){
		if(s[i]!=s[i+1]) res++;
		else {
			ans.push_back(s[i]);
			ans.push_back(s[i+1]);
		}
	}
	int cnt=1;
	for(int i=1;i<ans.size();i++){
		if(ans[i]!=ans[i-1]) {
			cnt++;
		}
	}
	cout<<res<<" "<<cnt<<endl;
	
}
signed main(){
	int _=1;
	cin>>_;
	while(_--) solve();
	return 0;
}

C. Tokitsukaze and Strange Inequality

这个题跟上次力扣周赛的有一题思想有点相同。四个点，枚举中间的那条边。

这个题目$O(n^2)$枚举b,c,然后就变成找 b 的右边有多少个比a[b] 的数 ，c 的左边有多少比 a[c] 小的数。

代码：

#include<bits/stdc++.h>
using namespace std;
// #define int long long
typedef long long ll;
typedef pair<int,int> PII;
const int N = 2e6 + 7;
const int mod = 1e9 + 7;
const int MOD = 998244353;
int sum1[5050][5050],sum2[5050][5050];
void solve(){
	int n;
	cin>>n;
	vector<int> a(n+1,0),v(n+1,0);
	for(int i=1;i<=n;i++) cin>>a[i];
	for(int i=1;i<=n;i++){
		int s=0;
		for(int j=1;j<=n;j++){
				sum1[i][j]=s;
				s+=v[j];
		}
		v[a[i]]=1;
	}
	for(int i=1;i<=n;i++) v[i]=0;
	for(int i=n;i;i--){
		int s=0;
		for(int j=1;j<=n;j++){
			sum2[i][j]=s;
			s+=v[j];
		}
		v[a[i]]=1;
	}
	ll res=0;//a b c d
	for(int i=2;i<=n;i++){//枚举中间两个数 b
		for(int j=i+1;j<=n-1;j++){//c
			res+=sum1[i][a[j]]*sum2[j][a[i]];//比a[b]大的数和比a[c]小的数
		}
	}
	cout<<res<<endl;
}
signed main(){
	int _=1;
	cin>>_;
	while(_--) solve();
	return 0;
}
